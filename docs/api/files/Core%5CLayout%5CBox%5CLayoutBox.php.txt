namespace WellCommerce\Core\Layout\Box;

/**
 * Class LayoutBox
 *
 * @package WellCommerce\Core\Layout
 * @author  Adam Piotrowski <adam@wellcommerce.org>
 */
class LayoutBox
{
    /**
     * @var int Layout box id
     */
    public $id;

    /**
     * @var array Layout box settings
     */
    public $settings;

    /**
     * @var string Layout box type
     */
    public $type;

    /**
     * @var string Layout box type
     */
    public $controller;

    /**
     * Constructor
     *
     * @param $box
     * @param $controller
     */
    public function __construct($box, $controller)
    {
        $this->id         = $box->id;
        $this->settings   = $box->settings;
        $this->type       = $box->type;
        $this->controller = $controller;
    }
}
