namespace WellCommerce\Core\Layout\Column;

use WellCommerce\Core\Layout\Box\LayoutBoxCollection;

/**
 * Class LayoutColumn
 *
 * @package WellCommerce\Core\Layout\Column
 * @author  Adam Piotrowski <adam@wellcommerce.org>
 */
class LayoutColumn
{
    /**
     * @var
     */
    public $width;

    /**
     * @var array
     */
    public $boxes = [];

    /**
     * Constructor
     *
     * @param       $width
     * @param       $template
     * @param array $boxes
     */
    public function __construct($width, LayoutBoxCollection $boxes)
    {
        $this->width    = $width;
        $this->boxes    = $boxes->all();
    }
}
